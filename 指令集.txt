全量训练及验证：
python tools/eval.py
python tools/train.py --epochs 7000 --batch_size 32 --cuda True
lite_data训练及验证：
python tools/train.py --data_dir lite_data --type lite --epochs 5 --batch_size 4 --no_pretrained
python tools/eval.py --data_dir lite_data --type lite

python tools/predict.py --data_type bottle --img-path images/demo0.png --dist_th 0.5

python deploy/export_model.py
python deploy/infer.py --data_type bottle --img-path images/demo0.png --dist_th 0.5
